custom_button = {
	name = upgrade_to_greatmarket_button
	tooltip = upgrade_to_greatmarket_button_tt
	potential = {
	}
	trigger = {
		FROM = {
			can_afford_to_upgrade_greatmarket = yes
			any_owned_province = {
				can_build = greatmarket
				OR = {
					has_building = marketplace
					has_building = trade_depot
				}
				NOT = {
					has_construction = any
				}
			}
		}
	}
	effect = {
		custom_tooltip = upgrade_to_greatmarket_effect_tt
		hidden_effect = {
			FROM = {
				upgrade_all_trade_buildings_to_greatmarkets = yes
			}
		}
	}
}
