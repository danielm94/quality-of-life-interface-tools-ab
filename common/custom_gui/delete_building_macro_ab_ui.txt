# Trade Buildings
custom_button = {
	name = building_macro_delete_greatmarket
	tooltip = building_macro_delete_greatmarket_tt
	potential = {
	}
	trigger = {
		is_province_not_whitelisted_delete_building_macro = yes
		FROM = {
			has_any_building_selected_for_deletion = no
			any_owned_province = {
				has_building = greatmarket
				is_province_not_whitelisted_delete_building_macro = yes
			}
		}
	}
	effect = {
		FROM = {
			hidden_effect = {
				set_country_flag = delete_building_macro_greatmarket
			}
			country_event = {
				id = delete_building_macro_events.1
			}
		}
	}
}
