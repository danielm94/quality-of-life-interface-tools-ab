custom_button = {
	name = building_macro_build_greatmarket
	tooltip = building_macro_build_greatmarket_tt
	potential = {
	}
	trigger = {
		FROM = {
			can_afford_to_build_greatmarket = yes
			any_owned_province = {
				can_build = greatmarket
			}
		}
	}
	effect = {
		if = {
			limit = {
				FROM = {
					has_country_flag = enable_smart_build_building_macro
				}
			}
			custom_tooltip = building_macro_strategy_highest_trade_value_tt
		}
		else = {
			custom_tooltip = building_macro_strategy_anywhere_possible_tt
		}
		custom_tooltip = building_macro_construction_logic_tt
		FROM = {
			build_buildings_macro_effect = {
				can_afford_to_build_trigger = can_afford_to_build_greatmarket
				building = greatmarket
				build_buildings_macro_strategy = build_buildings_highest_trade_value
			}
		}
	}
}
