#SCOPE = COUNTRY
calculate_greatmarket_build_cost = {
	export_to_variable = {
		which = country_build_cost
		value = modifier:build_cost
	}
	calculate_building_action_cost = {
		base_cost = 600
		var_name = greatmarket_build_cost
		build_cost_var = country_build_cost
	}
}
